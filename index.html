                        Merge1( PriorityQueue H1, PriorityQueue H2 )
                        {
                            /* single node */
                            if ( H1->Left == NULL ) 	
                                H1->Left = H2;
                            else {
                                /* Step 1 & 2 */
                                H1->Right = Merge( H1->Right, H2 ); 
                                if ( H1->Left->Npl < H1->Right->Npl )
                                    SwapChildren( H1 );	/* Step 3 */
                                H1->Npl = H1->Right->Npl + 1;
                            }
                            return H1;
                        }
